import copy
from problem import Problem
from sliding_tile_state import SlidingTileState
#

SlidingTileProblem inherits from Problem.

class SlidingTileProblem(Problem):
#

Constructor for SlidingTileProblem.

Given a description of the sliding tile puzzle, i.e. i is the number of rows, and j is the number of columns, and initial_state is a SlidingTileState describing the initial state of the puzzle, construct a new SlidingTileProblem.

    def __init__(self, i=3, j=3, initial_state=None):
#

Setup SlidingTileProblem like a Problem.

        super().__init__(initial_state=initial_state)
#

Set the number of rows and columns for the problem.

        self.num_rows = i
        self.num_cols = j
#

Setup a goal state.

        self.goal = SlidingTileState(i=i, j=j)
#

If an initial state wasn't defined, setup an initial state equal to the goal state.

        if not initial_state:
            self.initial_state = SlidingTileState(i=i, j=j)

        return
#

actions returns a list of possible actions that can be taken in state.

    def actions(self, state=None):
        actions = []
#

Test each action so see if we can take it in state.

        if state.can_move(state.UP):
            actions.append(state.UP)
        if state.can_move(state.DOWN):
            actions.append(state.DOWN)
        if state.can_move(state.LEFT):
            actions.append(state.LEFT)
        if state.can_move(state.RIGHT):
            actions.append(state.RIGHT)

        return actions
#

goal_test returns True if state is a goal state and False if it is not.

    def goal_test(self, state=None):
        return self.goal == state
#

result returns a SlidingTileState 

    def result(self, state=None, action=None):
#

Make a copy of state to return.

        resulting_state = copy.deepcopy(state)
#

Take action in state. Note that if the move cannot be done in state, then resulting_state is unmodified.

        resulting_state.move(action)

        return resulting_state
#

step_cost computes a cost of doing action in from_state to result in to_state.

For the sliding tile puzzle, every swap has a step cost of 1. But if action is a null action, or action can't be taken in from_state, then the step cost is zero.

    def step_cost(self, from_state=None, action=None, to_state=None):
        if None == action or not from_state.can_move(action):
            return 0
        else:
            return 1
#

get_num_rows returns the number of rows the puzzle has in the problem.

    def get_num_rows(self):
        return self.num_rows
#

get_num_cols returns the number of columns the puzzle has in the problem.

    def get_num_cols(self):
        return self.num_cols
