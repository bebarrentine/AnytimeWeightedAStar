import unittest

from node import Node
# 
class TestNode(unittest.TestCase):
# 
    def setUp(self):
        self.parent = Node(path_cost=1)
        self.action = 1
        self.state = 2
        self.path_cost = 3
        self.node = Node(
            action=self.action,
            path_cost=self.path_cost,
            parent=self.parent,
            state=self.state
        )
# 
    def test_get_parent(self):
        self.assertEqual(self.node.get_parent(), self.parent)
# 
    def test_get_state(self):
        self.assertEqual(self.node.get_state(), self.state)
# 
    def test_get_path_cost(self):
        self.assertEqual(self.node.get_path_cost(), self.path_cost)
# 
    def test_get_action(self):
        self.assertEqual(self.node.get_action(), self.action)
